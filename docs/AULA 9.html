library(tidyverse)
library(readxl)
data_mg <- read_excel("dados-diversos.xlsx")

dat2 <- data_mg |> 
  group_by(trat) |> 
  summarise (mean_comp = mean(comp),
             sd_comp = sd(comp),
             var_comp = var(comp),
             n = n(),
             se_comp = sd_comp / sqrt(n-1),
             ci = se_comp * qt(0.025, df = 9))
